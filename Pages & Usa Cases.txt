2.1 -  Review List
// It’s checks user’s review about it availability and submits it
//	$s_id : Student’s unique id
SELECT r.review_text, r.course_rating, r.term, s.name, s.surname
FROM reviews r, courses c,students s
WHERE r.cdn = c.cdn
AND r.term = c.term
AND c.creator_id = '$s_id'


3.1 - Search Schedule
//  Searches semester schedules with keywords and orders results by likes
//	$name_key : Keyword for Name	$descr_key : Keyword for Description
SELECT ss.`name`, ss.`description`, ss.`likes`, stu.`name`, stu.`surname`, stu.`mail`
FROM `semester_schedules` AS ss, `students` AS stu
WHERE ss.`name` LIKE '%$name_key%'
AND ss.`description` LIKE '%$descr_key%'
AND stu.`s_id` = ss.`creator_id`
ORDER BY ss.`likes` DESC

3.2 - Search Long Term Schedule
//  Searches long term schedules with keywords and orders results by likes
//	$name_key : Keyword for Name	$descr_key : Keyword for Description
SELECT lts.`name`, lts.`description`, lts.`likes`, stu.`name`, stu.`surname`, stu.`mail`
FROM `long_term_schedules` AS lts, `students` AS stu
WHERE lts.`name` LIKE '%$name_key%'
AND lts.`description` LIKE '%$descr_key%'
AND stu.`s_id` = lts.`creator_id`
ORDER BY lts.`likes` DESC

3.3 - Assign to a Semester Schedule
// Assigns a semester schedule to a user
//	$s_id : Student’s unique id		$ss_id : Semester Schedule’s unique id
INSERT INTO `s_has_ss` (`s_id`, `ss_id`) VALUES ('$s_id', '$ss_id')

3.4 - Assign to a Long Term Schedule

//	$s_id : Student’s unique id		$lts_id : Long Term Schedule’s unique id
INSERT INTO `s_has_lts` (`s_id`, `lts_id`) VALUES ('$s_id', '$lts_id')


4.1 - Creating Semester Schedule
// Creates a schedule for 1 semester for the user
//	$name : Schedule’s Name		$descr : Description
//	$creator_id : Foreign Key of Student’s Id who creates it
INSERT INTO `semester_schedules`(`name`, `description`, `creator_id`)
VALUES (`$name`, `$descr`, `$creator_id`)

4.2 - Creating Long Term Schedule
// Creates a long term schedule for user
//	$name : First Name			$descr : Description
//	$creator_id : Foreign Key of Student’s Id who creates it
INSERT INTO `long_term_schedules`(`name`, `description`, `creator_id`)
VALUES (`$creator_id`, `$name`, `$description`)
